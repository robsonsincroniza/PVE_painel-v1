function updateMetasTab() { const tbody = document.getElementById('metasBody'); if(!tbody) return; tbody.innerHTML = ''; let countAltaIntensidade = 0; let countRefinamento = 0; let sortedData = [...currentData].sort((a, b) => a.geral - b.geral); // FUNÇÃO AUXILIAR: Transforma a nota no crachá do Nível Macro (1, 2, 3, 4) com cores const getBadgeNivelMacro = (score) => { if (!score || score === 0) return `-`; if (score <= 2.50) return `Nível 1`; if (score <= 3.25) return `Nível 2`; if (score <= 3.70) return `Nível 3`; return `Nível 4`; }; sortedData.forEach(row => { const tr = document.createElement('tr'); tr.className = 'border-b border-slate-100 hover:bg-slate-50 transition-colors'; const local = row.uf !== 'Outro' ? `${row.municipio} (${row.uf})` : row.municipio; const subInfo = getSubLevelInfo(row.geral); let effortColor = 'bg-slate-50 text-slate-600'; let targetText = subInfo.m500 || ''; if (targetText.includes('7') || targetText.includes('8') || targetText.includes('9') || targetText.includes('10')) { effortColor = 'bg-orange-50 text-orange-700 font-bold'; countAltaIntensidade++; } else { effortColor = 'bg-emerald-50 text-emerald-700 font-bold'; countRefinamento++; } // Cálculo das competências mais fáceis de evoluir de nível const cityComps = []; for(let i=1; i<=8; i++) { if(row[`gesc_c${i}`] > 0) cityComps.push({name: `GESC C${i}`, val: row[`gesc_c${i}`]}); if(row[`gedu_c${i}`] > 0) cityComps.push({name: `GEDU C${i}`, val: row[`gedu_c${i}`]}); } for(let i=1; i<=5; i++) { if(row[`mob_c${i}`] > 0) cityComps.push({name: `MOB C${i}`, val: row[`mob_c${i}`]}); } cityComps.forEach(c => { if (c.val <= 2.50) { c.nextLevel = 2.51; c.dist = 2.51 - c.val; c.levelTarget = 2; } else if (c.val <= 3.25) { c.nextLevel = 3.26; c.dist = 3.26 - c.val; c.levelTarget = 3; } else if (c.val <= 3.70) { c.nextLevel = 3.71; c.dist = 3.71 - c.val; c.levelTarget = 4; } else { c.nextLevel = 4.0; c.dist = 999; c.levelTarget = 4; } }); cityComps.sort((a,b) => a.dist - b.dist); let match100 = (subInfo.m100 || '').match(/\d+/); let n100 = match100 ? parseInt(match100[0]) : 0; let match300 = (subInfo.m300 || '').match(/\d+/); let n300 = match300 ? parseInt(match300[0]) : 0; let match500 = (subInfo.m500 || '').match(/\d+/); let n500 = match500 ? parseInt(match500[0]) : 0; const tooltip100 = buildTooltipText('META 100', n100, cityComps); const tooltip300 = buildTooltipText('META 300', n300, cityComps); const tooltip500 = buildTooltipText('META 500', n500, cityComps); // AQUI: Aplica o crachá visual do Nível Macro (1, 2, 3 ou 4) na 2ª coluna let badgeNivel = getBadgeNivelMacro(row.geral); tr.innerHTML = ` ${local} ${badgeNivel} ${subInfo.m100 || '-'} ${subInfo.m300 || '-'}
${subInfo.m500 || '-'} Ver Focos
`; tbody.appendChild(tr); }); const elTotal = document.getElementById('metaTotalMun'); if (elTotal) elTotal.innerText = currentData.length; const elAlta = document.getElementById('metaAltaIntensidade'); if (elAlta) elAlta.innerText = countAltaIntensidade; const elRef = document.getElementById('metaRefinamento'); if (elRef) elRef.innerText = countRefinamento; try { if (chartInstances.metasDoughnut) { chartInstances.metasDoughnut.data.datasets[0].data = [countAltaIntensidade, countRefinamento]; chartInstances.metasDoughnut.update(); } if (chartInstances.miniMetas) { chartInstances.miniMetas.data.datasets[0].data = [countAltaIntensidade, countRefinamento]; chartInstances.miniMetas.update(); } } catch(e) { console.error("Metas Chart Error", e); } const summaryHtml = (countAltaIntensidade === 0 && countRefinamento === 0) ? "Sem dados de esforço disponíveis." : `${countAltaIntensidade} Alta Intensidade  |  ${countRefinamento} Refinamento`; const doughnutSummary = document.getElementById('metasDoughnutSummary'); if (doughnutSummary) doughnutSummary.innerHTML = summaryHtml; const miniSummary = document.getElementById('miniMetasSummary'); if (miniSummary) miniSummary.innerHTML = summaryHtml; }
